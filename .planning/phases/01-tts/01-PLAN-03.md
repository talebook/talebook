"""TTS 引擎抽象层测试"""
import pytest
from webserver.services.tts_engine import BaseTTSEngine, TTSEngineRegistry


class TestBaseTTSEngine:
    """BaseTTSEngine 抽象基类测试"""

    def test_base_is_abstract(self):
        """验证 BaseTTSEngine 是抽象类"""
        with pytest.raises(TypeError):
            BaseTTSEngine()


class MockEngine(BaseTTSEngine):
    """测试用 Mock 引擎"""
    KEY = "test_mock"

    def synthesize(self, text, output_path, **kwargs):
        return True

    def get_voices(self):
        return [{"id": "test", "lang": "en", "name": "Test"}]


class TestTTSEngineRegistry:
    """TTS 引擎注册表测试"""

    def setup_method(self):
        """每个测试前清空注册表"""
        TTSEngineRegistry._engines = {}

    def test_register_engine(self):
        """测试引擎注册"""
        engine = MockEngine()
        TTSEngineRegistry.register(engine)

        assert "test_mock" in TTSEngineRegistry.list_engines()

    def test_get_registered_engine(self):
        """测试获取已注册引擎"""
        engine = MockEngine()
        TTSEngineRegistry.register(engine)

        retrieved = TTSEngineRegistry.get("test_mock")
        assert retrieved is not None
        assert retrieved.KEY == "test_mock"

    def test_get_unregistered_engine(self):
        """测试获取未注册引擎返回 None"""
        retrieved = TTSEngineRegistry.get("nonexistent")
        assert retrieved is None

    def test_get_default_no_engines(self):
        """测试无引擎时 get_default 返回 None"""
        default = TTSEngineRegistry.get_default()
        assert default is None

    def test_get_default_with_engines(self):
        """测试有引擎时 get_default 返回第一个可用引擎"""
        engine1 = MockEngine()
        engine1.KEY = "first"
        engine2 = MockEngine()
        engine2.KEY = "second"

        TTSEngineRegistry.register(engine1)
        TTSEngineRegistry.register(engine2)

        default = TTSEngineRegistry.get_default()
        assert default is not None

    def test_list_engines_empty(self):
        """测试空注册表"""
        engines = TTSEngineRegistry.list_engines()
        assert engines == []

    def test_list_engines_multiple(self):
        """测试多引擎注册"""
        for key in ["engine1", "engine2", "engine3"]:
            e = MockEngine()
            e.KEY = key
            TTSEngineRegistry.register(e)

        engines = TTSEngineRegistry.list_engines()
        assert len(engines) == 3
        assert "engine1" in engines
        assert "engine2" in engines
        assert "engine3" in engines


class TestMockEngine:
    """Mock 引擎功能测试"""

    def test_synthesize_creates_file(self, mock_tts_engine, temp_audio_dir):

"""对话识别测试"""
import pytest
from webserver.services.tts import TTSService


class TestDialogueExtractor:
    """对话提取器测试"""

    @pytest.fixture
    def service(self):
        """创建 TTSService 实例"""
        return TTSService()

    def test_extract_english_double_quotes(self, service):
        """测试英文双引号提取"""
        text = 'He said: "Hello, World!"'
        dialogues = service._extract_dialogues(text)

        assert len(dialogues) >= 1
        texts = [d['text'] for d in dialogues]
        assert "Hello, World!" in texts

    def test_extract_english_single_quotes(self, service):
        """测试英文单引号提取"""
        text = "She said: 'Goodbye, Friend!'"
        dialogues = service._extract_dialogues(text)

        texts = [d['text'] for d in dialogues]
        assert "Goodbye, Friend!" in texts

    def test_extract_chinese_double_quotes(self, service):
        """测试中文双引号提取"""
        text = '他说：「你好，世界！」'
        dialogues = service._extract_dialogues(text)

        texts = [d['text'] for d in dialogues]
        assert any("你好" in t and "世界" in t for t in texts)

    def test_extract_chinese_single_quotes(self, service):
        """测试中文单引号提取"""
        text = "『你好』"
        dialogues = service._extract_dialogues(text)

        assert len(dialogues) >= 1

    def test_extract_multiple_dialogues(self, service):
        """测试多对话提取"""
        text = '"First" and "Second" and "Third"'
        dialogues = service._extract_dialogues(text)

        assert len(dialogues) >= 3
        texts = [d['text'] for d in dialogues]
        assert "First" in texts
        assert "Second" in texts
        assert "Third" in texts

    def test_extract_dialogues_sorted_by_position(self, service):
        """测试对话按位置排序"""
        text = '"A" xxx "B" yyy "C"'
        dialogues = service._extract_dialogues(text)

        positions = [d['start'] for d in dialogues]
        assert positions == sorted(positions), "Dialogues should be sorted by position"

    def test_filter_too_short(self, service):
        """测试过滤太短的文本"""
        text = '""'  # 太短
        dialogues = service._extract_dialogues(text)

        # 过滤后可能为空或很少
        for d in dialogues:
            assert len(d['text']) >= 5

    def test_filter_too_long(self, service):
        """测试过滤太长的文本"""
        text = '"' + "a" * 2000 + '"'  # 太长
        dialogues = service._extract_dialogues(text)

        for d in dialogues:
            assert len(d['text']) <= 1000

    def test_clean_dialogue_text(self, service):
        """测试对话文本清理"""
        # 测试基本清理
        text = "  Hello World  "
        cleaned = service._clean_dialogue_text(text)
        assert cleaned == "Hello World"

    def test_dialogue_id_sequential(self, service):
        """测试对话 ID 连续性"""
        text = '"A" "B" "C" "D"'
        dialogues = service._extract_dialogues(text)

        ids = [d['id'] for d in dialogues]
        assert ids == list(range(len(dialogues))), "IDs should be sequential"

    def test_no_dialogues(self, service):
        """测试无对话的情况"""
        text = "This is plain text without any quotes."
        dialogues = service._extract_dialogues(text)

        # 可能没有匹配，但不应该报错
        assert isinstance(dialogues, list)

    def test_mixed_quotes(self, service):
        """测试混合引号"""
        text = '''"English" and 「中文」 and 'single' '''
        dialogues = service._extract_dialogues(text)

        assert len(dialogues) >= 3

        """测试 ffmpeg 可用性"""
        result = subprocess.run(
            ['ffmpeg', '-version'],
            capture_output=True
        )
        assert result.returncode == 0

    def test_ffmpeg_mp3_encoding(self, temp_audio_dir):
        """测试 MP3 编码"""
        # 创建测试 WAV
        import os
        wav_path = f"{temp_audio_dir}/test.wav"
        mp3_path = f"{temp_audio_dir}/test.mp3"

        # 生成简单的测试音频（使用 ffmpeg 内置生成）
        subprocess.run([
            'ffmpeg',
            '-f', 'lavfi',
            '-i', 'sine=frequency=440:duration=1',
            wav_path,
            '-y'
        ], check=True, capture_output=True)

        # 转换为 MP3
        result = subprocess.run([
            'ffmpeg',
            '-i', wav_path,
            '-codec:a', 'libmp3lame',
            '-q:a', '2',
            mp3_path,
            '-y'
        ], check=True, capture_output=True)

        assert os.path.exists(mp3_path)
